Simple graphs

Create simple charts very easily. Just pick parent DOM element, put your data into an array, and run the library.
<!--We'll load the library's styles and scripts -->
<link href="../dist/d3.SimpleGraphs.css" rel="stylesheet">
<script src="../dist/lib/d3.v3.min.js" type="text/javascript"></script>
<script src="../dist/d3.SimpleGraphs.js" type="text/javascript"></script>
 
<!-- Optionally we can style the charts' containers-->
<style>
      .graph_container {
        height: 350px;
        width: 80%;
                margin: 15px 10%;
      }
</style>


<!-- Then we crete the charts' containers -->
<div id='graph1' class='graph_container'></div>
<div id='graph2' class='graph_container'></div>
<div id='graph3' class='graph_container'></div>
<div id='graph4' class='graph_container'></div>


<!-- Then we set the data for the charts-->
<script>
//These are two different sets of data
var positiveData = [25,23,21,17,14,11,10,12,14,17,20,23];
//Every graph will use either ont or the other set
var negPosData = [20,19,17,13,-2,-3,-7,8,9,13,15,18];
        
var keys = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"];
</script>

<!-- Finally, we start the four charts -->
<script>
d3.simpleGraphs.bars(document.getElementById('graph1'), keys, positiveData, {});

d3.simpleGraphs.bars(document.getElementById('graph2'), keys, negPosData, {});

d3.simpleGraphs.line(document.getElementById('graph3'), keys, negPosData, {});

d3.simpleGraphs.area(document.getElementById('graph4'), keys, positiveData, {});
</script>
